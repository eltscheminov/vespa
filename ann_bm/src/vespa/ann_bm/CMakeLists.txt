install(DIRECTORY DESTINATION share/ann_bm)

vespa_add_library(vespa_ann_bm
  ALLOW_UNRESOLVED_SYMBOLS
  SOURCES
  vespa_ann_bm.cpp

  INSTALL share/ann_bm
  DEPENDS
  pybind11::pybind11
)

if (TARGET pybind11::lto)
  target_link_libraries(vespa_ann_bm PRIVATE pybind11::module pybind11::lto)
else()
  target_link_libraries(vespa_ann_bm PRIVATE pybind11::module)
endif()

if (COMMAND pybind11_extension)
  pybind11_extension(vespa_ann_bm)
else()
  set_target_properties(vespa_ann_bm PROPERTIES PREFIX "${PYTHON_MODULE_PREFIX}")
  set_target_properties(vespa_ann_bm PROPERTIES SUFFIX "${PYTHON_MODULE_EXTENSION}")
endif()

set_target_properties(vespa_ann_bm PROPERTIES CXX_VISIBILITY_PRESET "hidden")
vespa_install_script(setup.py share/ann_bm)
